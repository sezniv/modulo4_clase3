id_novedad serial primary key, 
cod_animal integer not null,
cod_categoria integer not null,
cod_encargados integer not null,
cod_area intenger not null,

fecha_novedad TIMESTAMP,
comentario_novedad VARCHAR(100)

FOREIGN KEY (cod_animal)

FOREIGN KEY (cod_area)
